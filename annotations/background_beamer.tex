\tikzset{pics/.cd,
    hexa/.style={code={
            \draw[very thin,opacity=0.15] (0,0) --
            ++(60:1) --
            ++(0:1) --
            ++(-60:1) --
            ++(240:1) --
            ++(180:1) --
            ++(120:1);
        }
    }
}

\tikzset{pics/.cd,
    colored hexa/.style={code={
            \draw[very thin, fill=colorbg, opacity=0.15] (0,0) --
            ++(60:1) --
            ++(0:1) --
            ++(-60:1) --
            ++(240:1) --
            ++(180:1) --
            ++(120:1);
        }
    }
}

\tikzset{pics/.cd,
    down hexa chain/.style={code={
            \foreach \i in {1,...,#1} {
                \draw (0, -{2*(\i-1)*sin(60)}) pic{hexa};
            }
        }
    }
}

\tikzset{pics/.cd,
    up hexa chain/.style={code={
            \foreach \i in {1,...,#1} {
                \draw (0, {2*(\i-1)*sin(60)}) pic{colored hexa};
            }
        }
    }
}
